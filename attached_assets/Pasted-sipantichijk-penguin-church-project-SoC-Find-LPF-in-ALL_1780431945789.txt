# main parsing routine
def parse_cmdline_options():

        # defaults (don't use argparse mechanism so that interoperability
        # with parse_ini works nicely)
        cmdline_defaults = { 'family':'', 'device':'', 'work_dir':'work_pnr', 'output_dir':'outflow' }

        parser = argparse.ArgumentParser(description='Efinity programmer command line runner', formatter_class=HelpFormatter)
        parser.add_argument('design', help='design name')
        parser.add_argument('--family', help='family')
        parser.add_argument('-d', '--device', help='device')
        parser.add_argument('--work_dir', help='work directory')
        parser.add_argument('--output_dir', help='output directory')
        parser.add_argument('--timeout', type=int, default=None, help='timeout in seconds, default no timeout')
        parser.add_argument('--opt', dest='opts', nargs='+', action='append', default=[], help='pass option string to efx_pgm directly')

        if INTERNAL: efx_run_internal.EfxInternalCmdLineOpts.add_efx_run_pgm_opts(parser)

        args = parser.parse_args()

        # handle defaults
        for key in cmdline_defaults:
                if getattr(args, key) is None:
                        setattr(args, key, cmdline_defaults[key])

        return args



def option_exists(option, option_list):

        if option_list is not None:
                for x in option_list:
                        for y in x:
                                if (option + '=') in y:
                                        return True
        return False
