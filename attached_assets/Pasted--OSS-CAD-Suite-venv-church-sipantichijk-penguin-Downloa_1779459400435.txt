class GPIORuleInstanceName(base_rule.Rule):
    """
    Check if the instance name is valid.

    Name is guaranted to be unique at create, edit and loading.
    """
    valid_name_pattern = re.compile(r"[a-zA-Z][\/\~0-9a-zA-Z\_\[\]\.]*$")

    def __init__(self):
        super().__init__()
        self.name = "gpio_rule_inst_name"

    def check(self, checker, design_block: GPIO):

        util.gen_util.mark_unused(checker)

        gpio = design_block

        if self.check_name_empty_or_invalid(gpio.name, "Instance name is empty."):
            return

        gpio_reg: Optional[GPIORegistry] = checker.design.gpio_reg
        assert gpio_reg is not None

        if gpio_reg.is_bus_name_used(gpio.name):
            self.error(f"GPIO name {gpio.name} is used.")
            return
