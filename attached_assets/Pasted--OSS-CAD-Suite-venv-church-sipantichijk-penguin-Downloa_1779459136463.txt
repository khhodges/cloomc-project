class GPIORuleResource(BaseRuleResource):
    """
    Check if the resource name is valid.
    """

    def __init__(self):
        super().__init__()
        self.name = "gpio_rule_resource"
        self.blk_display_name = 'GPIO'

    def get_usable_resources(self, checker: GPIOChecker):
        if checker.gpiodev_service is not None:
            return checker.gpiodev_service.get_usable_instance_names()
        else:
            return []

class GPIORuleInputMode(base_rule.Rule):
    """
    Check if input mode config are valid
    """

    def __init__(self):
        super().__init__()
        self.name = "gpio_rule_input_mode"

    def check(self, checker, design_block):

        util.gen_util.mark_unused(checker)

        gpio = design_block

        if gpio.mode == gpio.PadModeType.input or gpio.mode == gpio.PadModeType.inout:

            input_cfg = gpio.input
            mode = "input used in inout" if gpio.mode == gpio.PadModeType.inout else "input"

            if input_cfg is None:
                # PT-1341: In old behaviour, inout mode does not check for input config,
                # if no input configured then just skip without error
                if not gpio.mode == gpio.PadModeType.inout:
                    self.error(
                        "For {} mode, input must be configured".format(mode))
                return

            if self.is_name_empty(input_cfg.name):
                # Allow empty input pin name in inout mode
                if not gpio.mode == gpio.PadModeType.inout:
                    self.error(
                        "For {} mode, input pin name must be configured".format(mode))
                    return
