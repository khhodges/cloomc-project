class GPIORuleHSIOStandardDifferential(base_rule.Rule):
    """
    Check that the GPIO with I/O std differential sstl/hstl is
    assigned to a p resource and that the n resource is not used
    """

    def __init__(self):
        super().__init__()
        self.name = "gpio_rule_io_standard_differential"

    def check(self, checker, design_block):

        gpio = design_block

        if self.is_name_empty(gpio.gpio_def):
            # Error registered by resource rule?
            return

        res_svc = checker.reg.res_svc

        # Get the hsio service from the registry
        if gpio.io_standard.find("Differential") != -1 and res_svc is not None:
            user_instances = res_svc.find_resource_user(gpio.gpio_def)

            if len(user_instances) > 1:
                # Get the name of the instances configured
                ins_name = []
                for inst in user_instances:
                    if inst is not None:
                        ins_name.append(inst.name)

                self.error("GPIO resource, {} with differential I/O standard was configured for multiple use: {}".format(
                    gpio.gpio_def, ",".join(ins_name)))
                return
            else:
                # PT-1265: Check if the other resource is bonded out
                res_list = res_svc.get_possible_hsio_gpio_resource_names(gpio.gpio_def)

                is_other_pin_available = True
                for res_name in res_list:
                    if res_name != gpio.gpio_def:
                        # Check if this is bonded out
                        if res_name not in checker.usable_hsio_gpio_ins_names:
                            is_other_pin_available = False
                            break

                if not is_other_pin_available:
                    self.error("Differential I/O standard is not valid on GPIO resource {} due to the corresponding N resource not bonded".format(
                        gpio.gpio_def))
                    return

            # Check that it is only on a p pad resource
            if not checker.hsiogpiodev_service.is_resource_on_p(gpio):
                self.error("Differential I/O standard can only be assigned to pad P resource")
                return
