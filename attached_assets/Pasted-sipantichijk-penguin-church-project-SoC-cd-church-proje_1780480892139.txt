# 2. Check P&R log for completion and EFX_RAM10 BRAM counts:
tail -30 work_pnr/pnr_new.log 2>/dev/null || tail -30 work_pnr/pnr.log

# 3. Try both USB interfaces — maybe TX is on ttyUSB3 not ttyUSB2:
stty -F /dev/ttyUSB3 115200 raw -echo 2>&1

# Power-cycle the board (unplug USB-C power, replug), then:
# listen on BOTH ports for 15 sec to catch whichever fires:
timeout 15 cat /dev/ttyUSB2 > /tmp/usb2.hex 2>/dev/null &
timeout 15 cat /dev/ttyUSB3 > /tmp/usb3.hex 2>/dev/null &
echo "Listening on both... power-cycle board NOW"
sleep 17
xxd /tmp/usb2.hex | head -10
echo "--- ttyUSB3 ---"
xxd /tmp/usb3.hex | head -10
