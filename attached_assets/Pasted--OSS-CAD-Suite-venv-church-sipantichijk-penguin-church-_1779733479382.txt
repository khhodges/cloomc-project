pad = (4 - len(data) % 4) % 4
data += b'\x00' * pad
words = [struct.unpack_from('<I', data, i)[0] for i in range(0, len(data), 4)]
words += [0] * (ROM - len(words))
words = words[:ROM]
lanes = [[], [], [], []]
for w in words:
    for l in range(4):
        lanes[l].append(format((w >> (l*8)) & 0xFF, '08b'))
prefix = 'EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol'
for l in range(4):
    fname = f'{prefix}{l}.bin'
    open(fname, 'w').write('\n'.join(lanes[l]) + '\n')
    nz = sum(1 for x in lanes[l] if x != '00000000')
    print(f'Wrote {fname}  ({nz} non-zero entries)')
print('Done.')
